Sets up the main loop for updating and rendering
             		                            
  start: ->
    unless @timer
      @timer = setInterval =>
        @update()
        @render()
      , Game.perMS
  
             		           
                               
                            
                 ¶               
               
Stops all updating and rendering
             		                            
  stop: ->
    clearInterval @timer
    @timer = null
  
             		           
                               
                            
                 ¶               
               
Update one frame
             		                            
  update: ->
    @stage.update()
  
             		           
                               
                            
                 ¶               
               
Render one frame
             		                            
  render: ->
    @canvas.clear()
    @stage.render(@ctx)
             		           
                               
                            
                 ¶               
               
Timing constants
             		                            
Game.fps    = 60
Game.perS   = 1/Game.fps
Game.perMS  = 1000/Game.fps
